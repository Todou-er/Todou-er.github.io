#include<stdio.h>
int main(void)
{
int n;
scanf(“%d”,&n);
printf(“%d”,n=f(n));
return 0;
}
int f(n)//递归 当n为1时返回1，不为1求下个数的阶乘
{
if(n==1) return 1;
else return n*f(n-1);
}
